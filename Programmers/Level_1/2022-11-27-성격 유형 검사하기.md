# 성격 유형 검사하기
## 문제 설명
https://school.programmers.co.kr/learn/courses/30/lessons/118666

## 제출
```swift
import Foundation

func solution(_ survey: [String], _ choices: [Int]) -> String {
    let score = [3, 2, 1, 0, 1, 2, 3]
    let defaultSurvey = ["RT", "CF", "JM", "AN"]
    var resultScore = [0, 0, 0, 0]
                  
    survey.enumerated().forEach { value in
        var 유형확인 = true
        
        let reverse = value.element.sorted {
            if $0 < $1 {
                유형확인 = false
                
                return true
            } else {
                return false
            }
        }.map { String($0) }.joined()
        
        if 유형확인 {
            let index = defaultSurvey.firstIndex(of: value.element)!
            
            if choices[value.offset] > 4 {
                resultScore[index] -= score[choices[value.offset] - 1]
            } else if choices[value.offset] < 4 {
                resultScore[index] += score[choices[value.offset] - 1]
            }
            
        } else {
            let index = defaultSurvey.firstIndex(of: reverse)!
            
            if choices[value.offset] > 4 {
                resultScore[index] += score[choices[value.offset] - 1]
            } else if choices[value.offset] < 4 {
                resultScore[index] -= score[choices[value.offset] - 1]
            }
        }
    }
    
    var result = ""
    
    resultScore.enumerated().forEach {
        if $0.element < 0 {
            result += defaultSurvey[$0.offset].last?.description ?? ""
        } else {
            result += defaultSurvey[$0.offset].first?.description ?? ""
        }
    }
    
    return result
}
```
