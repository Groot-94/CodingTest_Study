# 푸드 파이트 대회
## 문제 설명
https://school.programmers.co.kr/learn/courses/30/lessons/134240

## 제출
```swift
import Foundation

func solution(_ food:[Int]) -> String {
    var result = [Int]()
    
    for index in food.indices {
        if index > 0 {
            var calorie = index
            if food[index] > 1 {
                for _ in 1...food[index]/2 {
                    result.append(calorie)
                }
            } else if food[index] != 1 {
                result.append(calorie)
            }
            calorie += 1
        }
    }
    
    result.append(0)
    
    let reverseResult = result.map { String($0) }.joined() +  result.filter { $0 != 0 }.reversed().map { String($0) }.joined()
    
    return reverseResult
}
```
