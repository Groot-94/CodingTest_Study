# 크레인 인형뽑기 게임
## 문제 설명
https://school.programmers.co.kr/learn/courses/30/lessons/64061

## 제출
```swift
import Foundation

func solution(_ board:[[Int]], _ moves:[Int]) -> Int {
    var count = 0
    var result = [Int]()
    var boarded = board
    
    for number in moves {
        let tuple = fickUp(row: number, board: boarded)
        result = result.filter { $0 != 0 }
        
        if tuple.0 != 0 {
            boarded = tuple.1
        }
        
        if result.count > 0 , tuple.0 != 0 {
            if result[result.count - 1] == tuple.0 {
                result[result.count - 1] = 0
                count += 2
            } else {
                result.append(tuple.0)
            }
        } else {
            result.append(tuple.0)
        }
    }
    
    return count
}

func fickUp(row: Int, board:[[Int]]) -> (Int, [[Int]])  {
    var boarded = board
    let row = row - 1
    
    for line in 0...board.count-1 {
        if board[line][row] != 0 {
            boarded[line][row] = 0
            
            return (board[line][row], boarded)
        }
    }
    
    return (0, boarded)
}
```
