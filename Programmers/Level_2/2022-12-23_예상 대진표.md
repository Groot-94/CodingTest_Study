# 예상 대진표
## 문제 설명
- https://school.programmers.co.kr/learn/courses/30/lessons/12985

## 제출
```swift
import Foundation

func solution(_ n: Int, _ a:Int, _ b:Int) -> Int {
    var answer = 0
    var left = a
    var right = b
    
    while true {
        left = left / 2 + left % 2
        right = right / 2 + right % 2
        answer += 1
        
        if left == right {
            break
        }
    }
    
    return answer
}
```
